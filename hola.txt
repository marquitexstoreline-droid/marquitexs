        // Image upload functionality
        function setupImageUpload(uploadElement, fileInput, previewContainer, previewImg, removeButton) {
            // Click on the upload area to trigger file input
            uploadElement.addEventListener('click', function(e) {
                if (e.target !== removeButton && e.target !== previewImg) {
                    fileInput.click();
                }
            });

            // Handle file selection
            fileInput.addEventListener('change', function(e) {
                const file = e.target.files[0];
                if (file) {
                    // Validate file type
                    const validTypes = ['image/jpeg', 'image/jpg', 'image/png', 'image/gif'];
                    if (!validTypes.includes(file.type)) {
                        alert('Por favor, selecciona una imagen válida (JPG, PNG, GIF)');
                        return;
                    }

                    // Validate file size (max 5MB)
                    if (file.size > 5 * 1024 * 1024) {
                        alert('La imagen es demasiado grande. Por favor, selecciona una imagen menor a 5MB');
                        return;
                    }

                    // Display preview
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        previewImg.src = e.target.result;
                        previewContainer.style.display = 'block';
                        uploadElement.style.display = 'none';
                    };
                    reader.readAsDataURL(file);
                }
            });

            // Remove image
            removeButton.addEventListener('click', function() {
                fileInput.value = '';
                previewContainer.style.display = 'none';
                uploadElement.style.display = 'block';
            });

            // Drag and drop functionality
            uploadElement.addEventListener('dragover', function(e) {
                e.preventDefault();
                e.stopPropagation();
                uploadElement.style.borderColor = '#e63946';
                uploadElement.style.backgroundColor = '#f8f9fa';
            });

            uploadElement.addEventListener('dragleave', function(e) {
                e.preventDefault();
                e.stopPropagation();
                uploadElement.style.borderColor = '#ddd';
                uploadElement.style.backgroundColor = 'transparent';
            });

            uploadElement.addEventListener('drop', function(e) {
                e.preventDefault();
                e.stopPropagation();
                uploadElement.style.borderColor = '#ddd';
                uploadElement.style.backgroundColor = 'transparent';

                const files = e.dataTransfer.files;
                if (files.length) {
                    fileInput.files = files;
                    // Trigger the change event manually
                    const event = new Event('change', { bubbles: true });
                    fileInput.dispatchEvent(event);
                }
            });
        }

        document.getElementById('product-form').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const productId = document.getElementById('product-id').value;
            const productImageInput = document.getElementById('product-image-input');
            const previewImg = document.getElementById('preview-img');
            
            // Get image URL (either from file upload or existing URL)
            let imageUrl = '';
            if (productImageInput.files.length > 0) {
                // In a real application, you would upload the file to a server
                // For this demo, we'll use the data URL from the preview
                imageUrl = previewImg.src;
            } else if (productId) {
                // Keep existing image for edit
                imageUrl = products.find(p => p.id === parseInt(productId)).image;
            } else {
                // Default image if no image uploaded for new product
                imageUrl = 'https://placehold.co/300x300/f8f9fa/1d3557?text=Product';
            }
            
            const productData = {
                name: document.getElementById('product-name').value,
                category: document.getElementById('product-category').value,
                price: parseFloat(document.getElementById('product-price').value),
                stock: parseInt(document.getElementById('product-stock').value),
                description: document.getElementById('product-description').value,
                image: imageUrl,
                badge: productId ? products.find(p => p.id === parseInt(productId)).badge : 'new'
            };
            
            if (productId) {
                // Edit existing product
                const index = products.findIndex(p => p.id === parseInt(productId));
                if (index !== -1) {
                    products[index] = { ...products[index], ...productData };
                }
            } else {
                // Add new product
                const newId = products.length > 0 ? Math.max(...products.map(p => p.id)) + 1 : 1;
                products.push({ id: newId, ...productData });
            }
            
            // Refresh the display
            loadAdminData();
            updateProductDisplay();
            hideProductForm();
            alert(productId ? 'Producto actualizado correctamente' : 'Producto añadido correctamente');
        });
